  ST7735_Init();

  // Start Listening on UART
  HAL_UART_Receive_IT(&huart2, &uart2_byte, 1);

  // Initial Screen

  ST7735_FillScreen(BLUE);
  ST7735_UpdateCharCursor(0,0);
  ST7735_PrintString("Terminal Active.");
  new_data_received = false;

  while (1)
  {
      // 1. Send the Poll Request
      Send_Webhook_Trigger("poll");

      // 2. The Smart Wait Loop (5 seconds total)
      // Iterates 50 times with 100ms delay to keep button responsive
      for(int i=0; i<50; i++) {

          HAL_Delay(100); // Wait 0.1 seconds

          // --- A. CHECK FOR INCOMING MAIL ---
          if (new_data_received == true) {
              new_data_received = false;

              if (strcmp(safe_buffer, "LivingRoomLightOn") == 0){
            	  HAL_GPIO_WritePin(LivingRoomLight_GPIO_Port, LivingRoomLight_Pin, GPIO_PIN_SET);

            	  ST7735_UpdateCharCursor(0, 20);
            	  ST7735_PrintString("Living Room Light On");
              }
              if (strcmp(safe_buffer, "LivingRoomLightOff") == 0){
            	  HAL_GPIO_WritePin(LivingRoomLight_GPIO_Port, LivingRoomLight_Pin, GPIO_PIN_RESET);

            	  ST7735_UpdateCharCursor(0, 20);
            	  ST7735_PrintString("Living Room Light Off");
			  }
              else {
            	  ST7735_UpdateCharCursor(0, 20);
            	  ST7735_PrintString("RX: ");
            	  ST7735_PrintString(safe_buffer);
              }

              break;
          }

          // --- B. CHECK BUTTON (PC13) ---
          // PC13 is Low (RESET) when pressed
          if (HAL_GPIO_ReadPin(GPIOC, GPIO_PIN_13) == GPIO_PIN_RESET) {

              // We CAN clear here because it happens rarely (only when you click)
              ST7735_FillScreen(BLUE);
              ST7735_UpdateCharCursor(0,0);
              ST7735_PrintString("Sending AI Req...");

              Send_Webhook_Trigger("ai");

              HAL_Delay(500); // Debounce

              // Reset text
              ST7735_FillScreen(BLUE);
              ST7735_UpdateCharCursor(0,0);
              ST7735_PrintString("Terminal Active.");

              break; // Restart loop immediately
          }
      }
  }
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }

// --- UART CALLBACK (INTERRUPT) ---
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart){
    if (huart == &huart2){
        if ((uart2_byte != '\r') && (uart2_byte != '\n') && (uart2_byte != '\0')){
            if(buffer_position < MAX_MESSAGE_SIZE - 1) {
                message[buffer_position] = uart2_byte;
                buffer_position++;
            }
        } else {
            message[buffer_position] = '\0';

            // Safe copy to avoid race conditions
            strncpy(safe_buffer, (char*)message, 99);
            safe_buffer[99] = '\0';
            new_data_received = true;

            memset(message, 0, MAX_MESSAGE_SIZE);
            buffer_position = 0;
        }
        HAL_UART_Receive_IT(&huart2, &uart2_byte, 1);
    }
}

// --- WEBHOOK TRIGGER ---
void Send_Webhook_Trigger(char* action) {
    char cmdBuffer[50];
    sprintf(cmdBuffer, "WEBHOOK:%s\r\n", action);
    HAL_UART_Transmit(&huart2, (uint8_t*)cmdBuffer, strlen(cmdBuffer), 100);
}

void ST7735_WriteData(uint8_t data[], uint16_t size) {
 HAL_GPIO_WritePin(DC_GPIO_Port, DC_Pin, GPIO_PIN_SET);
 HAL_GPIO_WritePin(CS_GPIO_Port, CS_Pin, GPIO_PIN_RESET);
 HAL_SPI_Transmit(&hspi1, data, size, HAL_MAX_DELAY);
 HAL_GPIO_WritePin(CS_GPIO_Port, CS_Pin, GPIO_PIN_SET);
}

void ST7735_WriteCommand(uint8_t cmd) {
 HAL_GPIO_WritePin(DC_GPIO_Port, DC_Pin, GPIO_PIN_RESET);
 HAL_GPIO_WritePin(CS_GPIO_Port, CS_Pin, GPIO_PIN_RESET);
 HAL_SPI_Transmit (&hspi1, &cmd, 1, HAL_MAX_DELAY);
 HAL_GPIO_WritePin(CS_GPIO_Port, CS_Pin, GPIO_PIN_SET);
}

void ST7735_Init(void) {
 HAL_GPIO_WritePin(RST_GPIO_Port, RST_Pin, GPIO_PIN_RESET);
 HAL_Delay(5);
 HAL_GPIO_WritePin(RST_GPIO_Port, RST_Pin, GPIO_PIN_SET);
 HAL_Delay(5);

 ST7735_WriteCommand(SWRESET);
 HAL_Delay(150);
 ST7735_WriteCommand(SLPOUT);
 HAL_Delay(150);
 ST7735_WriteCommand(COLMOD);
 uint8_t colorMode = 0x05;
 ST7735_WriteData(&colorMode, 1);
 ST7735_WriteCommand(MADCTL);
 uint8_t accessMode = 0xC8;
 ST7735_WriteData(&accessMode, 1);
 ST7735_WriteCommand(DISPON);
 HAL_Delay(10);
}

void ST7735_SetAddressWindow(uint16_t xStart, uint16_t yStart, uint16_t xEnd, uint16_t yEnd) {
 if ((xStart > xEnd) || (yStart > yEnd) || (xEnd >= NUMCOLS) || (yEnd >= NUMROWS)) return;
	 uint8_t address[4];
	 ST7735_WriteCommand(CASET);

	 address[0] = xStart >> 8;
	 address[1] = xStart & 0xFF;
	 address[2] = xEnd >> 8;
	 address[3] = xEnd & 0xFF;

	 ST7735_WriteData(address, 4);
	 ST7735_WriteCommand(RASET);

	 address[0] = yStart >> 8;
	 address[1] = yStart & 0xFF;
	 address[2] = yEnd >> 8;
	 address[3] = yEnd & 0xFF;

	 ST7735_WriteData(address, 4);
	 ST7735_WriteCommand(RAMWR);
}

void ST7735_DrawPixel(uint8_t x, uint8_t y, uint16_t color) {
 if ((x >= NUMCOLS) || (y >= NUMROWS)) return;
 ST7735_SetAddressWindow(x, y, x, y);
 uint8_t pixelData[2] = { color >> 8, color & 0xFF };
 ST7735_WriteData(pixelData, 2);
}

void ST7735_FillScreen(uint16_t color) {
    ST7735_SetAddressWindow(0, 0, NUMCOLS - 1, NUMROWS - 1);
    for (uint8_t y = 0; y < NUMROWS; y++) {
        for (uint8_t x = 0; x < NUMCOLS; x++) {
            ST7735_DrawPixel(x, y, color);
        }
    }
}

void ST7735_UpdateCharCursor(uint16_t x, uint16_t y){
 cursor_x = x;
 cursor_y = y;
 if (cursor_x > (NUMCOLS-8)){
 cursor_x = 0;
 cursor_y += 8;
 }
 if (cursor_y > (NUMROWS-9)){
 cursor_x = 0;
 cursor_y = 0;
 }
}

void ST7735_PrintChar(char character) {
 if ((character < ' ') || (character > '~')) return;
 int fontIndex = character - 0x20;
 const uint8_t* characterData = fontTable[fontIndex];
 ST7735_SetAddressWindow(cursor_x, cursor_y, cursor_x + 7, cursor_y + 7);
 for (int row = 0; row < 8; row++) {
 uint8_t rowData = characterData[row];
 for (int col = 7; col >= 0; col--) {
 uint16_t color;
 if ((rowData & (1 << col)) == 0) color = WHITE;
 else color = BLACK;
 uint8_t pixelData[2] = { color >> 8, color & 0xFF };
 ST7735_WriteData(pixelData, 2);
 }
 }
 ST7735_UpdateCharCursor(cursor_x+8, cursor_y);
}

void ST7735_PrintString(char* string) {
 for (int i = 0; string[i] != '\0'; i++){
 ST7735_PrintChar(string[i]);
 }
}
